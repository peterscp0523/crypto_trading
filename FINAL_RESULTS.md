def auto_rebalance():
    # 현재 비중 계산
    current = calculate_current_weights()

    # 목표 비중 (균등 or 성과 기반)
    target = {
        'BTC': 0.20,
        'ETH': 0.20,
        'SOL': 0.20,
        'XRP': 0.20,
        'ADA': 0.20
    }

    # 차이만큼 매매
    for coin in coins:
        diff = target[coin] - current[coin]
        if abs(diff) > 0.05:  # 5% 이상 차이
            adjust(coin, diff)
